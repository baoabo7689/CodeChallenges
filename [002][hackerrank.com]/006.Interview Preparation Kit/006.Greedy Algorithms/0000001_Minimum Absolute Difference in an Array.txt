class Solution {

    // Complete the minimumAbsoluteDifference function below.
    static int minimumAbsoluteDifference(int[] arr) {
        var lst = arr.ToList();
        lst.Sort();
        long result = 2000000000;
        for(var i=1; i<lst.Count; i++) {
            result = Math.Min(Math.Abs(lst[i] - lst[i-1]), result);
        }

        return (int)result;
    }
